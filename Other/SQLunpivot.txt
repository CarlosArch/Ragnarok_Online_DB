SELECT
    MOBS.ID
   ,MOBS.NAME
   ,MOBS.LEVEL
   ,MOBS.HP
   ,MOBS.BASE_EXP
   ,MOBS.JOB_EXP
   ,MOBS.MVP_EXP
   ,IIF(MOBS.MVP_EXP > 0,1,0) AS MOB_IS_BOSS
   ,MOBS.PLANT

   ,'MVP_DROP_1' AS DROP_TYPE
   ,MOBS.MVP_DROP_ID_1
   ,ITEMS.NAME
   ,ITEMS.WEIGHT
   ,ITEMS.SELL_PRICE
   ,MOBS.MVP_DROP_PERCENTAGE_1
FROM
    MOBS
    LEFT JOIN ITEMS ON
        MOBS.MVP_DROP_ID_1 = ITEMS.ID

UNION ALL
